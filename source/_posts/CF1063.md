### B

[https://codeforces.com/contest/2163/problem/B](https://codeforces.com/contest/2163/problem/B) #思维题 #分类讨论 #观察 #构造

首先是题面一定要读仔细，漏读条件就会导致大量时间的浪费，在这道题就是后面才看到 $x_{i}=0$ 的时候，$s_{i}$ 可以是任何值

其次要注意观察数据范围，题面先是提醒最多五次操作，又强调你不必要用最少次数去做，其实意思已经非常明显了，那题面就是明着告诉你，你现在要构造一个 5 次的操作，让所有的情况下，这个操作都可以符合条件

要这样其实就是要把 s 几乎全为 1，因为根据规则不难看出，边界不可能为 1，1 和 n 所在的下标也不可能为 1，这些地方如果为 1，就直接输出-1

那么考虑怎么样才能让一次操作尽可能地多地让 0 变为 1，也就是选择最小值和最大值，所以我们注意到 1 和 n 两个特殊的值

```text
#include <bits/stdc++.h>
#define mod 100000009
#define int long long
using namespace std;
using PII = pair<int, int>;

void solve() {
    int n; cin >> n;
    vector<int> p(n);
    for (auto& it : p) cin >> it;
    string x; cin >> x;

    for (int i = 0; i < n; i++) {
        if (x[i] == '1' && (i == 0 || i == n - 1 || p[i] == 1 || p[i] == n)) {
            cout << -1 << "\n";
            return;
        }
    }

    int a = find(p.begin(), p.end(), 1) - p.begin() + 1;
    int b = find(p.begin(), p.end(), n) - p.begin() + 1;

    if (a > b) {
        swap(a, b);
    }

    cout << 5 << "\n";
    cout << 1 << " " << a << "\n";
    cout << 1 << " " << b << "\n";
    cout << a << " " << b << "\n";
    cout << a << " " << n << "\n";
    cout << b << " " << n << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

### C

[https://codeforces.com/contest/2163/problem/C](https://codeforces.com/contest/2163/problem/C) #前缀和 #后缀和 #合并区间 #去重

这道题主要卡我的点在于如何去重

不难注意到，我们实际上只有 n 条不同的路，对于每一条路我们都可以得到最小的区间，但这里面存在重合，那么怎么去重呢？

我们可以先获得每一条路的最小区间，所有满足条件的区间需要至少满足至少包括这些区间的一个

我们先假定以某一个 l 点出发，只要找到满足题意的最小的 R，再遍历一遍所有的 l 点，把满足条件的数量相加就是答案了

那么怎么找出满足题意的最小 R 呢？

这里需要用后缀最小值来实现

```text
#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (n); ++i)
#define int long long
using namespace std;
using ll = long long;
using PII = pair<int, int>;
using PLL = pair<long long, long long>;

void solve() {
    int n; cin >> n;
    vector a(2, vector<int>(n));
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < n; j++) {
            cin >> a[i][j];
            a[i][j]--;
        }
    }

    auto pmin = a[0], pmax = a[0], smin = a[1], smax = a[1];
    for (int i = 1; i < n; i++) {
        pmin[i] = min(pmin[i - 1], pmin[i]);
        pmax[i] = max(pmax[i - 1], pmax[i]);
    }
    for (int i = n - 2; i >= 0; i--) {
        smin[i] = min(smin[i], smin[i + 1]);
        smax[i] = max(smax[i], smax[i + 1]);
    };

    vector<int> f(2 * n, 2 * n);
    for (int i = 0; i < n; i++) {
        int L = min(pmin[i], smin[i]);
        int R = max(pmax[i], smax[i]);
        f[L] = min(f[L], R);
    }

    for (int i = 2 * n - 2; i >= 0; i--) {
        f[i] = min(f[i], f[i + 1]);
    }

    int ans = 0;
    for (int i = 0; i < 2 * n; i++) {
        ans += 2 * n - f[i];
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
