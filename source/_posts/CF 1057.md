### B

#思维

观察规律，把 xyz 并排拆成 2 进制看，注意到同一位上如果有两个数为 1，那么第三个数应该也为 1，所以把同一位的数相加，如果等于 2，那么就是不可能的

### 解题小技巧

```cpp
// 1 取当前位
int bit_X = x & 1;
// 2 移位
x >>= 1;
void solve() {
    int x, y, z;
    cin >> x >> y >> z;

    while (x > 0 || y > 0 || z > 0) {
        int bit_x = x & 1;
        int bit_y = y & 1;
        int bit_z = z & 1;

        if ((bit_x + bit_y + bit_z) == 2) {
            cout << "NO" << endl;
            return;
        }

        x >>= 1;
        y >>= 1;
        z >>= 1;
    }

    cout << "YES" << endl;
}
```

### C

#模拟 #几何

很容易注意到偶数个一定要贪心用掉，其他的单个棍子保留，然后判断是否可以使用， 有一个坑点就是没有判断单个棍子的长度是否长于其他棍子的总长

* * *

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

void solve() {
    int n;
    cin >> n;
    map<int, int> mp;
    for (int i = 0; i < n; ++i) {
        int x;
        cin >> x;
        mp[x]++;
    }

    int pp = 0;
    int mlen = 0;
    vector<int> singles;

    for (auto const& [len, count] : mp) {
        if (count >= 2) {
            pp += len * (count / 2 * 2);
            mlen = max(mlen, len);
        }
        if (count % 2 == 1) {
            singles.push_back(len);
        }
    }

    sort(singles.rbegin(), singles.rend());

    int ans = 0;

    if (pp > 0 && 2 * mlen < pp) {
        ans = pp;
    }

    for (int s : singles) {
        if ((pp / max(1LL, mlen) / 2 * 2) + 1 < 3 && pp==0){
            continue;
        }

        int cp = pp + s;
        int cmx = max(mlen, s);

        if (2 * cmx < cp) {
            ans = max(ans, cp);
            break;
        }
    }

    if (singles.size() >= 2) {
        for (size_t i = 0; i + 1 < singles.size(); ++i) {
            int s1 = singles[i];
            int s2 = singles[i+1];

            int p = pp + s1 + s2;
            int cml = max(mlen, s1);

            if (2 * cml < p) {
                ans = max(ans, p);
                break;
            }
        }
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```

## D

[https://codeforces.com/contest/2153/problem/D](https://codeforces.com/contest/2153/problem/D)

#DP #线性DP #分块

启发性的循环写法：用一次遍历把相邻两个数进行交换

```cpp
for (int i = 2; i <= n; i++) {
        swap(a[i], a[i-1]);
    }
// 交换后就可以实现全部数进行左移一位
```

分块的思想：要使得差值最小，那么就要把数组分为 3 个数或者 2 个数一个小组，每个小组计算，再累加到答案上，那么题目就转化为了一个选或者不选的问题，于是就想到了使用 DP 计算，对于每一个下标 i，i 可能是和前面两个数一组，也可能是和前面一个数一组，线性 DP，可以通过O（n）的时间复杂度得到答案

```cpp
#include <bits/stdc++.h>
#define rep(i, n) for(int i = 0; i < n; i++)
#define int long long
using namespace std;
using PII = pair<int, int>;
using ll = long long;
constexpr ll inf = 0x3f3f3f3f3f3f3f3f;

ll cost(int x, int y) { return abs(x - y); }

ll cost(int x, int y, int z) {
    if (x > y) {
        swap(x, y);
    }
    if (y > z) {
        swap(y, z);
    }
    if (x > y) {
        swap(x, y);
    }
    return z - x;
}

void solve() {
    int n; cin >> n;
    vector<int> a(n+1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    int ans = inf;
    for (int cyc = 0; cyc < 4; cyc++) {
        vector<int> dp(n+1, inf);
        dp[0] = 0;
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i-2] + cost(a[i-1], a[i]);
            if (i >= 3) {
                dp[i] = min(dp[i], dp[i-3] + cost(a[i-2], a[i-1], a[i]));
            }
        }
        ans = min(ans, dp[n]);

        for (int i = 2; i <= n; i++) {
            swap(a[i], a[i-1]);
        }
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
