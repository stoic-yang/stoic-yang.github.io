### C Clamped Sequence

## 暴力

这里非常的小丑，首先应该注意到，数据范围是 5e3，是可以支持 $O (n^2)$ 的时间复杂度的，所以其实可以直接暴力枚举

### 去重复技巧：

```cpp
// 先排序
sort(a.begin(), a.end());
// 使用 erase + unique配合，注意最后好药加上一个a.end()，参数
a.erase(unique(a.begin(), a.end()), a.end());
#include <bits/stdc++.h>
#define int long long
using namespace std;

void solve() {
    int n, d;
    cin >> n >> d;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    auto calc = [&](int l, int r) {
        vector<int> b(n);
        for (int i = 0; i < n; ++i) {
            if (a[i] < l) b[i] = l;
            else if (a[i] > r) b[i] = r;
            else b[i] = a[i];
        }
        int diff = 0;
        for (int i = 0; i < n - 1; ++i) {
            diff += abs(b[i] - b[i + 1]);
        }
        return diff;
    };

    vector<int> elements = a;
    sort(elements.begin(), elements.end());
    elements.erase(unique(elements.begin(), elements.end()), elements.end());

    int ans = 0;
    for (int val : elements) {
        ans = max(ans, calc(val, val + d));
        ans = max(ans, calc(val - d, val));
    }

    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    // cin >> T;
    while (T--) {
        solve();
    }

    return 0;
}
```

### 概率计算

`uniformly at random`，计算这个概率是不能通过计算总数量来得到，还必须要通过每一层的概率相加得到，可能是我的概率学的不好，不太理解两着有什么区别，有机会我要再找几道概率的题目写一写，看到底怎么回事。

另一个启发的点就是，在计算这类概率的时候，直接全部开 double 就完事了，最后要是还乘以一个 $1.0$ 反而麻烦

### dfs （递归）设计

我自己写的 dfs 简直就是一坨，最后甚至 TLE 了，现在想来应该就是我每次到达终点都要计算一次 A 队和 B 队的生存状态，这个时间开销的巨大的，另外，我还传入了两个复制的数组作为参数，这就更加 sb，纯纯冗余的代码。

dfs 的参数应该带有两类信息，一个是可以帮助确定递归边界的信息（i，n，在本题目中为队伍 1 和队伍 2 的死亡数量），还有一个是递归过程中需要传递到终点的参数。

写 dfs 时，先考虑递归的边界，先把边界条件写好，再写递归主逻辑

**在 dfs 中，每次调用递归后一定要进行复原!!!**，这个题目，我就是忘记把一个比较外层的攻击次数复原，导致了很多次的模拟错误。

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
using PII = pair<int, int>;

void solve() {
    int n, m;
    cin >> n >> m;
    vector<int> a(n), v1(n), b(m), v2(m);
    for (int i = 0; i < n; i++) cin >> a[i], v1[i] = a[i];
    for (int i = 0; i < m; i++) cin >> b[i], v2[i] = b[i];
    vector<int> atk1(n), atk2(m);
    double res1= 0, res2 = 0, res3 = 0;
    auto dfs = [&](auto self, int k, int s1, int s2, double v) {
        if (s1 == n || s2 == m) {
            if (s1 == n && s2 == m) res3 += v;
            else if (s1 == n) res2 += v;
            else if (s2 == m) res1 += v;
            return;
        }

        if (k % 2 == 0) {
            int p = -1;
            for (int i = 0; i < n; i++) {
                if (a[i] > 0 && (p == -1 || atk1[i] < atk1[p])) {
                    p = i;
                }
            }
            atk1[p]++;
            double cnt = 0;
            for (int i = 0; i < m; i++) {
                if (b[i] > 0) {
                    cnt++;
                }
            }

            for (int i = 0; i < m; i++) {
                if (b[i] > 0) {
                    a[p] -= v2[i], b[i] -= v1[p];
                    if (a[p] <= 0 && b[i] <= 0) {
                        self(self, k ^ 1, s1 + 1, s2 + 1, v / cnt);
                    }
                    else if (a[p] <= 0) {
                        self(self, k ^ 1, s1 + 1, s2, v / cnt);
                    }
                    else if (b[i] <= 0) {
                        self(self, k ^ 1, s1, s2 + 1, v / cnt);
                    }
                    else {
                        self(self, k ^ 1, s1, s2, v / cnt);
                    }
                    a[p] += v2[i], b[i] += v1[p];
                }
            }
            atk1[p]--;
        }
        else {
            int p = -1;
            for (int i = 0; i < m; i++) {
                if (b[i] > 0 && (p == -1 || atk2[i] < atk2[p])) {
                    p = i;
                }
            }
            atk2[p]++;
            double cnt = 0;
            for (int i = 0; i < n; i++) {
                if (a[i] > 0) {
                    cnt++;
                }
            }

            for (int i = 0; i < n; i++) {
                if (a[i] > 0) {
                    b[p] -= v1[i], a[i] -= v2[p];
                    if (a[i] <= 0 && b[p] <= 0) {
                        self(self, k ^ 1, s1 + 1, s2 + 1, v / cnt);
                    }
                    else if (a[i] <= 0) {
                        self(self, k ^ 1, s1 + 1, s2, v / cnt);
                    }
                    else if (b[p] <= 0) {
                        self(self, k ^ 1, s1, s2 + 1, v / cnt);
                    }
                    else {
                        self(self, k ^ 1, s1, s2, v / cnt);
                    }
                    b[p] += v1[i], a[i] += v2[p];
                }
            }
            atk2[p]--;
        }
    };

    if (n < m) dfs(dfs, 1, 0, 0, 1);
    else if (n > m) dfs(dfs, 0, 0, 0, 1);
    else dfs(dfs, 0, 0, 0, 1), dfs(dfs, 1, 0, 0, 1);

    cout << fixed << setprecision(10);
    double s = res1 + res2 + res3;
    cout << res1 / s << "\n" << res2 / s << "\n" << res3 / s << "\n";
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    // cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}
```
